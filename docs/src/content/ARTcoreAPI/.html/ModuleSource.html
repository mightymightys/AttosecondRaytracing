 34def Source(SourceProperties):
 35    SourcePosition = np.array([0, 0, 0])
 36    SourceDirection = np.array([1, 0, 0])
 37    Divergence = SourceProperties["Divergence"]
 38    SourceSize = SourceProperties["SourceSize"]
 39    Wavelength = SourceProperties["Wavelength"]
 40    RayNumber =  SourceProperties["NumberRays"]
 41    if Divergence == 0:
 42        if SourceSize != 0:
 43            radius = SourceSize / 2
 44            SourceRayList = PlaneWaveDisk(SourcePosition, SourceDirection, radius, RayNumber, Wavelength=Wavelength)
 45        else:
 46            # If the source is of size 0 and divergence 0, it's a single ray
 47            SourceRayList = [mray.Ray(SourcePosition, SourceDirection, Wavelength=Wavelength)]
 48    else:
 49        if SourceSize == 0:
 50            SourceRayList = PointSource(
 51                SourcePosition, SourceDirection, Divergence, RayNumber, Wavelength=Wavelength
 52            )
 53        else:
 54            SourceRayList = ExtendedSource(
 55                SourcePosition, SourceDirection, SourceSize, Divergence, RayNumber, Wavelength=Wavelength
 56            )
 57
 58    SourceRayList = ApplyGaussianIntensityToRayList(
 59        SourceRayList, 1 / np.e**2
 60    )  # Intensity drops to 1/e^2 at edge of ray-bundle

 62def _Cone(Angle: float, NbRays: int, Wavelength=None):
 63    """
 64    Return a list of rays filling a cone according to Vogel's spiral.
 65
 66    Parameters
 67    ----------
 68        Angle : float
 69            Cone apex *half*-angle in radians.
 70
 71        NbRays : int
 72            Number of rays in the list.
 73
 74        Wavelength : float, optional
 75            Wavelength of the light, set as attribute to all rays.
 76    """
 77    Height = 1
 78    Radius = Height * np.tan(Angle)
 79
 80    MatrixXY = mgeo.SpiralVogel(NbRays, Radius)
 81
 82    RayList = []
 83
 84    for k in range(NbRays):
 85        x = MatrixXY[k, 0]
 86        y = MatrixXY[k, 1]
 87        RayList.append(mray.Ray(np.array([0, 0, 0]), np.array([x, y, Height]), Number=k, Wavelength=Wavelength))
 88
 89    return RayList

 93def PointSource(S: np.ndarray, Axis: np.ndarray, Divergence: float, NbRays: int, Wavelength=None):
 94    """
 95    Return a list of rays simulating rays from a point source.
 96
 97    ![Illustration of a point source.](pointsource.svg)
 98
 99    Parameters
100    ----------
101        S : np.ndarray
102            Coordinate vector of the source point (in mm).
103
104        Axis : np.ndarray
105            Axis vector of the cone (main direction of the rays)
106
107        Divergence : float
108            Cone apex *half*-angle in *radians* (similar to gaussian beam).
109
110        NbRays : int
111            Number of rays in the list.
112
113        Wavelength : float, optional
114            Wavelength of the light, set as attribute to all rays.
115
116    """
117    RayList = _Cone(Divergence, NbRays, Wavelength=Wavelength)
118    RayList = mgeo.RotationRayList(RayList, np.array([0, 0, 1]), Axis)
119    RayList = mgeo.TranslationRayList(RayList, S)
120    return RayList

124def ExtendedSource(S: np.ndarray, Axis: np.ndarray, Diameter: float, Divergence: float, NbRays: int, Wavelength=None):
125    """
126    Return a list of rays simulating rays from an extended array of point sources, distributed over a disk with Diameter.
127
128    Parameters
129    ----------
130        S : np.ndarray
131            Coordinate vector of the source point (in mm).
132
133        Axis : np.ndarray
134            Axis vector of the cone (main direction of the rays)
135
136        Diameter : float
137            Diameter of the extended source
138
139        Divergence : float
140            Cone apex *half*-angle in *radians* (similar to gaussian beam).
141
142        NbRays : int
143            Number of rays in the list.
144
145        Wavelength : float, optional
146            Wavelength of the light, set as attribute to all rays.
147
148    """
149    NbPointSources = max(5, int(250 * Diameter))
150    NbPointSources = min(NbPointSources, 100)
151
152    MatrixXY = mgeo.SpiralVogel(NbPointSources, Diameter / 2)  # the positions of the point sources
153
154    NbRaysPerPointSource = max(100, int(NbRays / NbPointSources))
155    RayList = []
156    PointSourceRayList = _Cone(Divergence, NbRaysPerPointSource, Wavelength=Wavelength)
157    for k in range(NbPointSources):
158        ShiftedPointSourceRayList = mgeo.TranslationRayList(PointSourceRayList, [MatrixXY[k, 0], MatrixXY[k, 1], 0])
159        for l in range(NbRaysPerPointSource):
160            ShiftedPointSourceRayList[l]._number += (
161                k * NbRaysPerPointSource
162            )  # we allow ourselves exceptionally to modify the "internal" _number attribute
163        RayList.extend(ShiftedPointSourceRayList)
164
165    RayList = mgeo.RotationRayList(RayList, np.array([0, 0, 1]), Axis)
166    RayList = mgeo.TranslationRayList(RayList, S)
167    return RayList

171def PlaneWaveDisk(Centre: np.ndarray, Axis: np.ndarray, Radius: float, NbRays: int, Wavelength=None):
172    """
173    Return a list of rays, all parallel and distributed over a round cross section according to a Vogel-spiral (see illustration). Simulating a 'round' collimated beam.
174
175    ![Illustration of a Vogel spiral.](doc_VogelSpiral.png)
176
177    Parameters
178    ----------
179        Centre : np.ndarray
180            Coordinate vector of the center of the source disk (in mm).
181
182        Axis : np.ndarray
183            Axis vector of the the plane wave (vector of all rays)
184
185        Radius : float
186            Radius of the source ray-bundle.
187
188        NbRays : int
189            Number of rays in the list.
190
191        Wavelength : float, optional
192            Wavelength of the light, set as attribute to all rays.
193
194    """
195    MatrixXY = mgeo.SpiralVogel(NbRays, Radius)
196    RayList = []
197    num = 0
198    for k in range(NbRays - 1):
199        x = MatrixXY[k, 0]
200        y = MatrixXY[k, 1]
201        RayList.append(mray.Ray(np.array([x, y, 0]), np.array([0, 0, 1]), Number=num, Wavelength=Wavelength))
202        num = num + 1
203    RayList = mgeo.RotationRayList(RayList, np.array([0, 0, 1]), Axis)
204    RayList = mgeo.TranslationRayList(RayList, Centre)
205    return RayList

209def PlaneWaveSquare(Centre: np.ndarray, Axis: np.ndarray, SideLength: float, NbRays: int, Wavelength=None):
210    """
211    Return a list of rays, all parallel and distributed over a square cross section awith SideLength. Simulating a 'square' collimated beam.
212
213    Parameters
214    ----------
215        Centre : np.ndarray
216            Coordinate vector of the center of the source disk (in mm).
217
218        Axis : np.ndarray
219            Axis vector of the the plane wave (vector of all rays)
220
221        SideLength : float
222            Side length of the square source ray-bundle.
223
224        NbRays : int
225            Number of rays in the list.
226
227        Wavelength : float, optional
228            Wavelength of the light, set as attribute to all rays.
229
230    """
231    RayList = []
232    x = np.linspace(-SideLength / 2, SideLength / 2, int(np.sqrt(NbRays)))
233    y = np.linspace(-SideLength / 2, SideLength / 2, int(np.sqrt(NbRays)))
234    RayList.append(mray.Ray(np.array([0, 0, 0]), np.array([0, 0, 1]), Number=0))
235    num = 1
236    for i in x:
237        for j in y:
238            if abs(x) > 1e-4 and abs(y) > 1e-4:
239                RayList.append(mray.Ray(np.array([i, j, 0]), np.array([0, 0, 1]), Number=num, Wavelength=Wavelength))
240                num += 1
241    RayList = mgeo.RotationRayList(RayList, np.array([0, 0, 1]), Axis)
242    RayList = mgeo.TranslationRayList(RayList, Centre)
243    return RayList

255def ApplyGaussianIntensityToRayList(RayList, IntensityFraction=1 / np.e**2):
256    """
257    Apply a Gaussain intensity profile to a ray list, with intensity =1 at the center and falling
258    to 'IntensityFraction' at the edge (arbitrary units).
259
260    Parameters
261    ----------
262        RayList : list of Ray-objects
263            Coordinate vector of the center of the source disk (in mm).
264
265        IntensityFraction : float, optional
266            Relative intensity in (0,1), reached at the edge of the ray bundle. Default is 1/e^2.
267
268    """
269    if IntensityFraction >= 1 or IntensityFraction <= 0:
270        # raise ValueError('IntensityFraction should be between 0 and 1!')
271        print(
272            "When applying a Gaussian intensity profile to a ray list, the IntensityFraction should be between 0 and 1! I'm setting it to 1/e^2."
273        )
274        IntensityFraction = 1 / np.e**2
275
276    Axis = mp.FindCentralRay(RayList).vector
277    Divergence = 0
278    for l in RayList:
279        # find the largest angle with the central ray out of all the rays in the list
280        Divergence = max(Divergence, mgeo.AngleBetweenTwoVectors(Axis, l.vector))
281
282    if (
283        Divergence > 1e-12
284    ):  # we're dealing with a point source and not a plane wave, so we can apply Gaussian intensity profile as function of angle
285        for k in RayList:
286            Angle = mgeo.AngleBetweenTwoVectors(Axis, k.vector)
287            Intensity = np.exp(-2 * (np.tan(Angle) / Divergence) ** 2 * -0.5 * np.log(IntensityFraction))
288            k.intensity = Intensity
289    else:  # otherwise we apply it as function of ray distance from the Axis
290        MaxDist = 0
291        for l in RayList:
292            MaxDist = max(MaxDist, np.linalg.norm(l.point))
293        for k in RayList:
294            Intensity = np.exp(-2 * (np.linalg.norm(k.point) / MaxDist) ** 2 * -0.5 * np.log(IntensityFraction))
295            k.intensity = Intensity
296
297    return RayList
