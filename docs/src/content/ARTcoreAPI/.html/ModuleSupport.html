Provides classes for different shapes of a support for optics.

A support fixes the spatial extent and outer shape of optics - think of it as a substrate. Technically, the support is a section of the x-y-plane in the element coordinate frame.

The support will become an attribute of Mirror or Mask objects.

Created in Sept 2020

@author: Anthony Guillaume and Stefan Haessler

489def find_hull(points):
490    # start from leftmost point
491    current_point = min(range(len(points)), key=lambda i: points[i][0])
492    # initialize hull with current point
493    hull = [current_point]
494    # initialize list of linked points
495    linked = []
496    # continue until all points have been linked
497    while len(linked) < len(points) - 1:
498        # initialize minimum distance and closest point
499        min_distance = math.inf
500        closest_point = None
501        # find closest unlinked point to current point
502        for i, point in enumerate(points):
503            if i not in linked:
504                distance = math.dist(points[current_point], point)
505                if distance < min_distance:
506                    min_distance = distance
507                    closest_point = i
508        # add closest point to hull and linked list
509        hull.append(closest_point)
510        linked.append(closest_point)
511        # update current point
512        current_point = closest_point
513    # add link between last point and first point
514    hull.append(hull[0])
515    # convert hull to a list of pairs of indices
516    indices = [[hull[i], hull[i + 1]] for i in range(len(hull) - 1)]
517    return indices

520def gen_rectangle_contour(dimX, dimY, NbPoints):
521    # calculate number of points on each side
522    nX = math.ceil(dimX / (dimX + dimY) * NbPoints)
523    nY = NbPoints - nX
524    # calculate distance between points on each side
525    dX = dimX / (nX - 1)
526    dY = dimY / (nY - 1)
527    # generate points on top side
528    top_points = [(i * dX - dimX / 2, dimY / 2) for i in range(nX)]
529    # generate points on right side
530    right_points = [(dimX / 2, dimY / 2 - i * dY) for i in range(1, nY)]
531    # generate points on bottom side
532    bottom_points = [(dimX / 2 - i * dX, -dimY / 2) for i in range(1, nX)]
533    # generate points on left side
534    left_points = [(-dimX / 2, -dimY / 2 + i * dY) for i in range(1, nY - 1)]
535    # concatenate points in counterclockwise order
536    points = top_points + right_points + bottom_points + left_points
537    return np.array(points)

540def gen_circle_contour(radius, NbPoints):
541    # calculate angle between points
542    if NbPoints == 0:
543        return np.array([])
544    else:    
545        angle = 2 * math.pi / NbPoints
546    # generate points
547    points = [(radius * math.cos(i * angle), radius * math.sin(i * angle)) for i in range(NbPoints)]
548    return np.array(points)

551def flatten_point_arrays(outer, holes=[], edges=False):
552    coords = [i for i in outer]
553    edges_list = []
554    offset = len(coords)
555    for arr in holes:
556        # add coordinates to list
557        coords.extend(arr)
558        # add edges to list
559        if edges:
560            n = arr.shape[0]
561            edge_indices = np.arange(n)
562            edge_indices += offset
563            edges_list += [edge_indices.T.tolist() + [offset]]
564        # update offset
565        offset += arr.shape[0]
566    if edges:
567        return coords, edges_list
568    else:
569        return coords

490def find_hull(points):
491    # start from leftmost point
492    current_point = min(range(len(points)), key=lambda i: points[i][0])
493    # initialize hull with current point
494    hull = [current_point]
495    # initialize list of linked points
496    linked = []
497    # continue until all points have been linked
498    while len(linked) < len(points) - 1:
499        # initialize minimum distance and closest point
500        min_distance = math.inf
501        closest_point = None
502        # find closest unlinked point to current point
503        for i, point in enumerate(points):
504            if i not in linked:
505                distance = math.dist(points[current_point], point)
506                if distance < min_distance:
507                    min_distance = distance
508                    closest_point = i
509        # add closest point to hull and linked list
510        hull.append(closest_point)
511        linked.append(closest_point)
512        # update current point
513        current_point = closest_point
514    # add link between last point and first point
515    hull.append(hull[0])
516    # convert hull to a list of pairs of indices
517    indices = [[hull[i], hull[i + 1]] for i in range(len(hull) - 1)]
518    return indices

521def gen_rectangle_contour(dimX, dimY, NbPoints):
522    # calculate number of points on each side
523    nX = math.ceil(dimX / (dimX + dimY) * NbPoints)
524    nY = NbPoints - nX
525    # calculate distance between points on each side
526    dX = dimX / (nX - 1)
527    dY = dimY / (nY - 1)
528    # generate points on top side
529    top_points = [(i * dX - dimX / 2, dimY / 2) for i in range(nX)]
530    # generate points on right side
531    right_points = [(dimX / 2, dimY / 2 - i * dY) for i in range(1, nY)]
532    # generate points on bottom side
533    bottom_points = [(dimX / 2 - i * dX, -dimY / 2) for i in range(1, nX)]
534    # generate points on left side
535    left_points = [(-dimX / 2, -dimY / 2 + i * dY) for i in range(1, nY - 1)]
536    # concatenate points in counterclockwise order
537    points = top_points + right_points + bottom_points + left_points
538    return np.array(points)
