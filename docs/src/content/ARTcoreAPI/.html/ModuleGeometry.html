 43def AngleBetweenTwoVectors(U, V):
 44    """Return the angle in radians between the vectors U and V ; formula from W.Kahan"""
 45    u = np.linalg.norm(U)
 46    v = np.linalg.norm(V)
 47    return 2 * np.arctan2(np.linalg.norm(U * v - V * u), np.linalg.norm(U * v + V * u))

 51def IntersectionLinePlane(A, u, P, n):
 52    """
 53    Return the intersection point between a line and A plane.
 54    A is a point of the line, u a vector of the line ; P is a point of the plane, n a normal vector
 55    Line's equation : OM = u*t + OA , t a real
 56    Plane's equation : n.OP - n.OM = 0
 57    """
 58    t = np.dot(n, -A + P) / np.dot(u, n)
 59    I = u * t + A
 60    return I

 71def SpiralVogel(NbPoint, Radius):
 72    """
 73    Return a NbPoint x 2 matrix of 2D points representative of Vogel's spiral with radius Radius
 74    """
 75
 76    GoldenAngle = np.pi * (3 - np.sqrt(5))
 77    r = np.sqrt(np.arange(NbPoint) / NbPoint) * Radius
 78
 79    theta = GoldenAngle * np.arange(NbPoint)
 80
 81    Matrix = np.zeros((NbPoint, 2))
 82    Matrix[:, 0] = np.cos(theta)
 83    Matrix[:, 1] = np.sin(theta)
 84    Matrix = Matrix * r.reshape((NbPoint, 1))
 85
 86    return Matrix

174def DiameterPointList(PointList):
175    """
176    Return the diameter of the smallest circle (for 2D points) or sphere (3D points) including all the points
177    """
178    if len(PointList) == 0:
179        return None
180
181    elif len(PointList[0]) == 2:
182        Xmax = PointList[0][0]
183        Xmin = PointList[0][0]
184        Ymax = PointList[0][1]
185        Ymin = PointList[0][1]
186
187        for k in PointList:
188            x = k[0]
189            y = k[1]
190
191            Xmax = max(x, Xmax)
192            Xmin = min(x, Xmin)
193            Ymax = max(y, Ymax)
194            Ymin = min(y, Ymin)
195
196        X = abs(Xmax - Xmin)
197        Y = abs(Ymax - Ymin)
198        Diameter = max(X, Y)
199
200        return Diameter
201
202    elif len(PointList[0]) == 3:
203        Xmax = PointList[0][0]
204        Xmin = PointList[0][0]
205        Ymax = PointList[0][1]
206        Ymin = PointList[0][1]
207        Zmax = PointList[0][2]
208        Zmin = PointList[0][2]
209
210        for k in PointList:
211            x = k[0]
212            y = k[1]
213            z = k[2]
214
215            Xmax = max(x, Xmax)
216            Xmin = min(x, Xmin)
217            Ymax = max(y, Ymax)
218            Ymin = min(y, Ymin)
219            Zmax = max(z, Zmax)
220            Zmin = min(z, Zmin)
221
222        X = abs(Xmax - Xmin)
223        Y = abs(Ymax - Ymin)
224        Z = abs(Zmax - Zmin)
225        Diameter = max(X, Y)
226        Diameter = max(Diameter, Z)
227
228        return Diameter

232def CentrePointList(PointList):
233    """
234    Shift all 2D-points in PointList so as to center the point-cloud on the origin [0,0].
235    """
236    ListCoordX = []
237    ListCoordY = []
238    appendX = ListCoordX.append
239    appendY = ListCoordY.append
240
241    for k in PointList:
242        appendX(k[0])
243        appendY(k[1])
244
245    CentreX = (np.amax(ListCoordX) + np.amin(ListCoordX)) * 0.5
246    CentreY = (np.amax(ListCoordY) + np.amin(ListCoordY)) * 0.5
247
248    PointListCentered = []
249    append = PointListCentered.append
250    for k in PointList:
251        x = k[0] - CentreX
252        y = k[1] - CentreY
253        append(np.array([x, y]))
254
255    return PointListCentered

331def RotationAroundAxis(Axis, Angle, Vector):
332    """Rotate Vector by Angle (in rad) around Axis"""
333    vec = np.quaternion(*Vector)
334    q = QRotationAroundAxis(Axis, Angle)
335    VectorPrime = q * vec * np.conjugate(q)
336    return VectorPrime.imag
337
338def QRotationAroundAxis(Axis, Angle):
339    """Return np.quaternion for rotation by Angle (in rad) around Axis"""
340    rot_axis = Normalize(np.array([0.0] + Axis))
341    axis_angle = (Angle * 0.5) * rot_axis
342    qlog = np.quaternion(*axis_angle)
343    q = np.exp(qlog)
344    return q

347def QRotationVector2Vector(Vector1, Vector2):
348    """
349    Return a possible quaternion (among many) that would rotate Vector1 into Vector2
350    """
351    Vector1 = Normalize(Vector1)
352    Vector2 = Normalize(Vector2)
353    a = np.cross(Vector1, Vector2)
354    return np.quaternion(1 + np.dot(Vector1, Vector2), *a).normalized()

356def QRotationVectorPair2VectorPair(InitialVector1, Vector1, InitialVector2, Vector2):
357    """
358    Return the only quaternion that rotates InitialVector1 to Vector1 and InitialVector2 to Vector2.
359    """
360    Vector1 = Normalize(Vector1)
361    Vector2 = Normalize(Vector2)
362    Vector3 = Normalize(np.cross(Vector1,Vector2))
363    InitialVector1 = Normalize(InitialVector1)
364    InitialVector2 = Normalize(InitialVector2)
365    InitialVector3 = Normalize(np.cross(InitialVector1,InitialVector2))
366    rot_2Initial = np.zeros((3,3))
367    rot_2Initial[:,0] = InitialVector1
368    rot_2Initial[:,1] = InitialVector2
369    rot_2Initial[:,2] = InitialVector3
370    rot_2Final = np.zeros((3,3))
371    rot_2Final[:,0] = Vector1
372    rot_2Final[:,1] = Vector2
373    rot_2Final[:,2] = Vector3
374    q2Init = quaternion.from_rotation_matrix(rot_2Initial)
375    q2Fin = quaternion.from_rotation_matrix(rot_2Final)
376    return (q2Fin/q2Init).normalized()

378def RotationPoint(Point, Axis1, Axis2):
379    """Transform vector Point such that Axis1 in the old coordinate frame becomes vector Axis 2 in the new coordinate system"""
380    Angle = AngleBetweenTwoVectors(Axis1, Axis2)
381    if abs(Angle) < 1e-10:
382        Pointprime = Point
383    elif abs(Angle - np.pi) < 1e-10:
384        Pointprime = -Point
385    else:
386        N = np.cross(Axis1, Axis2)
387        Pointprime = RotationAroundAxis(N, Angle, Point)
388    return Pointprime

413    """Like RotationPoint but with a Ray object"""
414    Rayprime = Ray.copy_ray()

458def normal_add(N1, N2):
459    """
460    Simple function that takes in two normal vectors of a deformation and calculates
461    the total normal vector if the two deformations were individually applied.
462    """
463    normal1 = N1 / np.linalg.norm(N1)
464    normal2 = N2 / np.linalg.norm(N2)
465    grad1X = -normal1[0] / normal1[2]
466    grad1Y = -normal1[1] / normal1[2]
467    grad2X = -normal2[0] / normal2[2]
468    grad2Y = -normal2[1] / normal2[2]
469    gradX = grad1X + grad2X
470    gradY = grad1Y + grad2Y
471    return np.array([-gradX, -gradY, 1])
