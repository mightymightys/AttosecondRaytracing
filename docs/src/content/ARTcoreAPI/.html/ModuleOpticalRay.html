 15    """

 21    def __init__(
 22        self,
 23        Point: np.ndarray,
 24        Vector: np.ndarray,
 25        Path=(0.0,),
 26        Number=None,
 27        Wavelength=None,
 28        Incidence=None,
 29        Intensity=None,
 30    ):
 31        """
 32        Parameters
 33        ----------
 34            Point : np.ndarray
 35                Point source of the ray..
 36
 37            Vector : np.ndarray
 38                Direction unit-vector of the ray..
 39
 40            Path : List of floats, optional TODO correct considering update to list
 41                Path length in mm covered by ray *before* its starting Point. The default is 0.
 42
 43            Number : int, optional
 44                Index needed to link the incident ray and the reflected ray. The default is None, but it should be set by afterwards setting the attribute Ray._number.!
 45
 46            Wavelength : float, optional
 47                Wavelength of the light in mm. The default is None.
 48
 49            Incidence : float, optional
 50                Incidence angle in radian of the ray on the optical element *from which it originates*.
 51                Gets calculated during ray tracing. The default is None. The source rays therefore have "None" incidence.
 52
 53            Intensity : float, optional
 54                Intensity, or more strictly speaking the fluence fraction in arb.u. carried by the ray. The default is None.
 55
 56        """
 57        self.point = Point  # the setter checks
 58        self.vector = Vector  # Not recommended to directly modify as you lose matrix 
 59        self._path = Path
 60        self._wavelength = Wavelength
 61        self._incidence = Incidence
 62        self._intensity = Intensity
 63        if type(Number) == int or Number is None:
 64            self._number = Number
 65        else:
 66            raise TypeError("Ray Number must be an integer.")
 67
 68    def __repr__(self):
 69        repr =  f"{type(self).__name__}("
 70        repr += f"Point={self.point.__repr__()}, Vector={self.vector.__repr__()}, Path={self.path.__repr__()}, "
 71        if hasattr(self, "_number"):
 72            repr += f"Number={self._number}, "
 73        if hasattr(self, "_number"):
 74            repr += f"Wavelength={self.wavelength}, "
 75        if hasattr(self, "_number"):
 76            repr += f"Incidence={self.incidence}, "
 77        if hasattr(self, "_number"):
 78            repr += f"Intensity={self.intensity}"
 79        repr += ")"
 80        return repr
