 26class Mask(moe.OpticalElement):
 27    """
 28    A mask: a plane surface of the shape of its [Support](ModuleSupport.html), which stops all rays that hit it,
 29    while all other rays continue their path unchanged. No diffraction effects.
 30
 31    Attributes
 32    ----------
 33        support : [Support](ModuleSupport.html)-object
 34
 35        type : str 'Mask'.
 36
 37    Methods
 38    ----------

 40
 41        get_centre()
 42
 43        get_grid3D(NbPoints)

 46
 47    def __init__(self, Support, **kwargs):
 48        """
 49        Parameters
 50        ----------
 51            Support : [Support](ModuleSupport.html)-object
 52        """
 53        self.type = "Mask"
 54        self.support = Support
 55        self._position = np.array([0.0, 0.0, 0.0])
 56        self.curvature = mmirror.Curvature.FLAT
 57        self._positionned = False
 58        if "position" in kwargs:
 59            NewPosition = kwargs["position"]
 60            if type(NewPosition) == np.ndarray and len(NewPosition) == 3:
 61                logger.debug("Defining position of spherical mirror during creation")
 62                self._position = NewPosition
 63                self._positionned = True
 64        self._orientation = np.quaternion(1)
 65        self._oriented = False
 66        if "orientation" in kwargs:
 67            NewOrientation = kwargs["orientation"]
 68            if type(NewOrientation) == np.quaternion:
 69                logger.debug("Defining orientation of spherical mirror during creation")
 70                self._orientation = NewOrientation
 71                self._oriented = True
 72        self.oe_frame_normal = np.array([0, 0, 1.0])
 73        self.oe_frame_majoraxis = np.array([1.0, 0, 0])
 74        super().__init__()
 75
 76    def __repr__(self):
 77        match type(self.support):
 78            case msup.SupportRound:
 79                shape = f"Round {self.type}"
 80                size = f"Radius: {self.support.radius}"
 81            case msup.SupportRectangle:
 82                shape = f"Rectangular {self.type}"
 83                size = f"Size: {self.support.dimX} x {self.support.dimY}"
 84            case msup.SupportRectangleHole:
 85                shape = f"Rectangular holey {self.type}"
 86                size = f"Size: {self.support.dimX} x {self.support.dimY}, hole radius: {self.support.radiushole}"
 87            case msup.SupportRectangleRectHole:
 88                shape = f"Rectangular holey {self.type} with rectangular hole"
 89                size = f"Size: {self.support.dimX} x {self.support.dimY}, hole size: {self.support.holeX} x {self.support.holeY}"
 90            case msup.SupportRoundHole:
 91                shape = f"Round holey {self.type}"
 92                size = f"Radius: {self.support.radius}, hole radius: {self.support.radiushole}"
 93            case _:
 94                shape = f"{self.type}"
 95                size = "Custom shape"
 96        pretty_str = f"{shape}\n"
 97        pretty_str+= f"\t-{size}\n\t-"
 98        if self._positionned:
 99            pretty_str+=f"Position: {self.position}\n"
100        else:
101            pretty_str+="Not positionned\n"
102        pretty_str+="\t-"
103        if self._oriented:
104            pretty_str+=f"Orientation: {self.orientation}\n"
105        else:
106            pretty_str+="Not oriented\n"
107        return pretty_str

110    def _get_intersection(self, Ray):
111        """Return the intersection point between Ray and the xy-plane,
112        where rays do NOT hit the mask's support"""
113        t = -Ray.point[2] / Ray.vector[2]
114        I = Ray.vector * t + Ray.point
115        if t > 0 and not self.support._IncludeSupport(I):
116            PointIntersection = I
117        else:
118            PointIntersection = None
119
120        return PointIntersection
121
122    def get_local_normal(self, Point):
123        """Return normal unit vector in point 'Point' on the mask."""
124        Normal = np.array([0, 0, 1])
125        return Normal
126
127    def get_centre(self):
128        """Return 3D coordinates of the point on the mask surface at the center of its support."""
129        return np.array([0, 0, 0])
130
131    def get_grid3D(self, NbPoint, **kwargs):
132        """
133        Returns list of numpy-arrays containing the 3D-coordinates of points of the mask surface,
134        sampling the support in a number NbPoints of points. 
135        """
136        E = "edges" in kwargs and kwargs["edges"]
137        ListCoordXYZ = []
138        contour = int(round(0.1 * NbPoint))
139        contours = self.support._Contour_points(contour, **kwargs)
140        if E:
141            contours, contour_edges = contours
142        ListCoordXY = contours + self.support._get_grid(NbPoint - contour)
143        for k in ListCoordXY:
144            z = 0
145            ListCoordXYZ.append(np.array([k[0], k[1], z]))
146        if E:
147            return ListCoordXYZ, contour_edges
148        return ListCoordXYZ

152def _TransmitMaskRay(Mask, PointMask, Ray):
153    """Returns the transmitted ray"""
154    PointRay = Ray.point
155    VectorRay = Ray.vector
156    NormalMask = Mask.get_local_normal(PointMask)
157
158    AngleIncidence = mgeo.AngleBetweenTwoVectors(VectorRay, NormalMask)
159    Path = Ray.path + (np.linalg.norm(PointMask - PointRay),)
160
161    RayTransmitted = Ray.copy_ray()
162    RayTransmitted.point = PointMask
163    RayTransmitted.vector = VectorRay
164    RayTransmitted.incidence = AngleIncidence
165    RayTransmitted.path = Path
166
167    return RayTransmitted

171def TransmitMaskRayList(Mask, RayList):
172    """
173    Returns the the transmitted rays that pass the mask for the list of
174    incident rays ListRay.
175
176    Rays that hit the support are not further propagated.
177
178    Updates the reflected rays' incidence angle and path.
179
180    Parameters
181    ----------
182        Mask : Mask-object
183
184        ListRay : list[Ray-object]
185
186    """
187    ListRayTransmitted = []
188    for Ray in RayList:
189        PointMask = Mask._get_intersection(Ray)
190
191        if PointMask is not None:
192            RayTransmitted = _TransmitMaskRay(Mask, PointMask, Ray)
193            ListRayTransmitted.append(RayTransmitted)
194
195    return ListRayTransmitted
