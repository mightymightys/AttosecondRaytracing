 24def SupportRound_ContourSupport(self, Figure):
 25    """Draw support contour in MirrorProjection plots."""
 26    axe = Figure.add_subplot(111, aspect="equal")
 27    axe.add_patch(patches.Circle((0, 0), self.radius, alpha=0.08))
 28    return axe

 76def _RenderOpticalElement(OE, OEpoints, draw_mesh = False):
 77    OpticPointList, edge_faces = OE.get_grid3D(OEpoints, edges=True)  # in the optic's coordinate system
 78    # transform OpticPointList into "lab-frame"
 79    OpticPointList = mgeo.TranslationPointList(OpticPointList, -OE.get_centre())
 80    # MirrorMajorAxisPrime = mgeo.RotationPoint(OE.majoraxis, OE.normal, np.array([0, 0, 1]))
 81    # OpticPointList = mgeo.RotationPointList(OpticPointList, np.array([1, 0, 0]), MirrorMajorAxisPrime)
 82    # OpticPointList = mgeo.RotationPointList(OpticPointList, np.array([0, 0, 1]), OE.normal)
 83    OpticPointList = mgeo.RotationPointListByQ(OpticPointList, OE.orientation)
 84    OpticPointList = mgeo.TranslationPointList(OpticPointList, OE.position)
 85    
 86    # shift optics surfaces a little back normally?
 87    if not draw_mesh:    
 88        OpticPointList = [point - OE.normal*0.5 for point in OpticPointList]
 89
 90    optic_pts = pv.PolyData(OpticPointList)
 91    tess = None
 92    if draw_mesh:
 93        pts_coord = pv.PolyData(OpticPointList)
 94        lines = list(
 95            itertools.chain.from_iterable([[[2, e[i], e[i + 1]] for i in range(len(e) - 1)] for e in edge_faces])
 96        )
 97        faces = list(itertools.chain.from_iterable([[len(i) - 1] + i[:-1] for i in edge_faces]))
 98        if lines == []:
 99            lines = [0]
100        if faces == []:
101            faces = [0]
102        edges = pv.PolyData(
103            OpticPointList,
104            lines=lines,
105            faces=faces,
106        )
107        # Can't figure out some edge cases such as when part of the support is outside of the mirror
108        tess = pts_coord.delaunay_2d(edge_source=edges)
109    return optic_pts, tess
