 34def _getDetectorPoints(RayListAnalysed, Detector) -> tuple[np.ndarray, np.ndarray, float, float]:
 35    """
 36    Prepare the ray impact points on the detector in a format used for the plotting,
 37    and along the way also calculate the "spotsize" of this point-cloud on the detector.
 38
 39    Parameters
 40    ----------
 41        RayListAnalysed : list(Ray)
 42            A list of objects of the ModuleOpticalRay.Ray-class.
 43
 44        Detector : Detector
 45            An object of the ModuleDetector.Detector-class.
 46
 47    Returns
 48    -------
 49        DectectorPoint2D_Xcoord : np.ndarray with x-coordinates
 50
 51        DectectorPoint2D_Ycoord : np.ndarray with y-coordinates
 52
 53        FocalSpotSize : float
 54
 55        SpotSizeSD : float
 56    """
 57
 58    DetectorPointList2DCentre = Detector.get_PointList2DCentre(RayListAnalysed)
 59    DectectorPoint2D_Xcoord = DetectorPointList2DCentre[:,0]*1e3  # mm to µm
 60    DectectorPoint2D_Ycoord = DetectorPointList2DCentre[:,1]*1e3
 61
 62    FocalSpotSize = mgeo.DiameterPointList(DetectorPointList2DCentre)
 63    SpotSizeSD = mp.StandardDeviation(DetectorPointList2DCentre)
 64    return DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD

 68def getETransmission(RayListIn, RayListOut) -> float:
 69    """
 70    Calculates the energy transmission from RayListIn to RayListOut in percent by summing up the
 71    intensity-property of the individual rays.
 72
 73    Parameters
 74    ----------
 75        RayListIn, RayListOut : list(Ray)
 76            Lists of objects of the ModuleOpticalRay.Ray-class.
 77
 78    Returns
 79    -------
 80        ETransmission : float
 81    """
 82    ETransmission = 100 * sum(Ray.intensity for Ray in RayListOut) / sum(Ray.intensity for Ray in RayListIn)
 83    return ETransmission

 87def GetResultSummary(Detector, RayListAnalysed, verbose=False):
 88    """
 89    Calculate and return FocalSpotSize-standard-deviation and Duration-standard-deviation
 90    for the given Detector and RayList.
 91    If verbose, then also print a summary of the results for the given Detector.
 92
 93    Parameters
 94    ----------
 95        Detector : Detector
 96            An object of the ModuleDetector.Detector-class.
 97
 98        RayListAnalysed : list(Ray)
 99            List of objects of the ModuleOpticalRay.Ray-class.
100
101        verbose : bool
102            Whether to print a result summary.
103
104    Returns
105    -------
106        FocalSpotSizeSD : float
107
108        DurationSD : float
109    """
110    DetectorPointList2DCentre = Detector.get_PointList2DCentre(RayListAnalysed)
111    FocalSpotSizeSD = mp.StandardDeviation(DetectorPointList2DCentre)
112    DelayList = Detector.get_Delays(RayListAnalysed)
113    DurationSD = mp.StandardDeviation(DelayList)
114
115    if verbose:
116        FocalSpotSize = mgeo.DiameterPointList(DetectorPointList2DCentre)
117        summarystring = (
118            "At the detector distance of "
119            + "{:.3f}".format(Detector.get_distance())
120            + " mm we get:\n"
121            + "Spatial std : "
122            + "{:.3f}".format(FocalSpotSizeSD * 1e3)
123            + " \u03BCm and min-max: "
124            + "{:.3f}".format(FocalSpotSize * 1e3)
125            + " \u03BCm\n"
126            + "Temporal std : "
127            + "{:.3e}".format(DurationSD)
128            + " fs and min-max : "
129            + "{:.3e}".format(max(DelayList) - min(DelayList))
130            + " fs"
131        )
132
133        print(summarystring)
134
135    return FocalSpotSizeSD, DurationSD

139def SpotDiagram(RayListAnalysed, Detector, DrawAiryAndFourier=False, ColorCoded=None) -> plt.Figure:
140    """
141    Produce a an interactive figure with the spot diagram resulting from the RayListAnalysed
142    hitting the Detector.
143    The detector distance can be shifted with the left-right cursor keys.
144    If DrawAiryAndFourier is True, a circle with the Airy-spot-size will be shown.
145    The 'spots' can optionally be color-coded by specifying ColorCoded
146    as ["Intensity","Incidence","Delay"].
147
148    Parameters
149    ----------
150        RayListAnalysed : list(Ray)
151            List of objects of the ModuleOpticalRay.Ray-class.
152
153        Detector : Detector
154            An object of the ModuleDetector.Detector-class.
155
156        DrawAiryAndFourier : bool, optional
157            Whether to draw a circle with the Airy-spot-size. The default is False.
158
159        ColorCoded : str, optional
160            Color-code the spots according to one of ["Intensity","Incidence","Delay"]. The default is None.
161
162    Returns
163    -------
164        fig : matlplotlib-figure-handle.
165            Shows the interactive figure.
166    """
167    NumericalAperture = mp.ReturnNumericalAperture(RayListAnalysed, 1)  # NA determined from final ray bundle
168    Wavelength = RayListAnalysed[0].wavelength
169    if DrawAiryAndFourier:
170        AiryRadius = mp.ReturnAiryRadius(Wavelength, NumericalAperture) * 1e3  # in µm
171    else:
172        AiryRadius = 0
173
174    DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD = _getDetectorPoints(
175        RayListAnalysed, Detector
176    )
177
178    if ColorCoded == "Intensity":
179        IntensityList = [k.intensity for k in RayListAnalysed]
180        z = np.asarray(IntensityList)
181        zlabel = "Intensity (arb.u.)"
182        title = "Intensity + Spot Diagram\n press left/right to move detector position"
183        addLine = ""
184    elif ColorCoded == "Incidence":
185        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # degree
186        z = np.asarray(IncidenceList)
187        zlabel = "Incidence angle (deg)"
188        title = "Ray Incidence + Spot Diagram\n press left/right to move detector position"
189        addLine = ""
190    elif ColorCoded == "Delay":
191        DelayList = Detector.get_Delays(RayListAnalysed)
192        DurationSD = mp.StandardDeviation(DelayList)
193        z = np.asarray(DelayList)
194        zlabel = "Delay (fs)"
195        title = "Delay + Spot Diagram\n press left/right to move detector position"
196        addLine = "\n" + "{:.2f}".format(DurationSD) + " fs SD"
197    else:
198        z = "red"
199        title = "Spot Diagram\n press left/right to move detector position"
200        addLine = ""
201
202    Dist = Detector.get_distance()
203    distStep = min(50, max(0.0005, round(FocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000))  # in mm
204
205    plt.ion()
206    fig, ax = plt.subplots()
207    if DrawAiryAndFourier:
208        theta = np.linspace(0, 2 * np.pi, 100)
209        x = AiryRadius * np.cos(theta)
210        y = AiryRadius * np.sin(theta)  #
211        ax.plot(x, y, c="black")
212
213    foo = ax.scatter(
214        DectectorPoint2D_Xcoord,
215        DectectorPoint2D_Ycoord,
216        c=z,
217        s=15,
218        label="{:.3f}".format(Dist) + " mm\n" + "{:.1f}".format(SpotSizeSD * 1e3) + " \u03BCm SD" + addLine,
219    )
220
221    axisLim = 1.1 * max(AiryRadius, 0.5 * FocalSpotSize * 1000)
222    ax.set_xlim(-axisLim, axisLim)
223    ax.set_ylim(-axisLim, axisLim)
224
225    if ColorCoded == "Intensity" or ColorCoded == "Incidence" or ColorCoded == "Delay":
226        cbar = fig.colorbar(foo)
227        cbar.set_label(zlabel)
228
229    ax.legend(loc="upper right")
230    ax.set_xlabel("X (µm)")
231    ax.set_ylabel("Y (µm)")
232    ax.set_title(title)
233    # ax.margins(x=0)
234
235    movingDetector = Detector.copy_detector()
236    movingDetector = Detector
237
238    def press(event):
239        nonlocal Dist, distStep, movingDetector, ColorCoded, zlabel, cbar
240        if event.key == "right":
241            movingDetector.shiftByDistance(distStep)
242            Dist += distStep
243        elif event.key == "left":
244            if Dist > 1.5 * distStep:
245                movingDetector.shiftByDistance(-distStep)
246                Dist -= distStep
247            else:
248                movingDetector.shiftToDistance(0.5 * distStep)
249                Dist = 0.5 * distStep
250        else:
251            return None
252        newDectectorPoint2D_Xcoord, newDectectorPoint2D_Ycoord, newFocalSpotSize, newSpotSizeSD = _getDetectorPoints(
253            RayListAnalysed, movingDetector
254        )
255        xy = foo.get_offsets()
256        xy[:, 0] = newDectectorPoint2D_Xcoord
257        xy[:, 1] = newDectectorPoint2D_Ycoord
258        foo.set_offsets(xy)
259
260        if ColorCoded == "Delay":
261            newDelayList = np.asarray(movingDetector.get_Delays(RayListAnalysed))
262            newDurationSD = mp.StandardDeviation(newDelayList)
263            newaddLine = "\n" + "{:.2f}".format(newDurationSD) + " fs SD"
264            foo.set_array(newDelayList)
265            foo.set_clim(min(newDelayList), max(newDelayList))
266            cbar.update_normal(foo)
267        else:
268            newaddLine = ""
269
270        foo.set_label(
271            "{:.3f}".format(Dist) + " mm\n" + "{:.1f}".format(newSpotSizeSD * 1e3) + " \u03BCm SD" + newaddLine
272        )
273        ax.legend(loc="upper right")
274
275        axisLim = 1.1 * max(AiryRadius, 0.5 * newFocalSpotSize * 1000)
276        ax.set_xlim(-axisLim, axisLim)
277        ax.set_ylim(-axisLim, axisLim)
278
279        distStep = min(
280            50, max(0.0005, round(newFocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000)
281        )  # in mm
282
283        fig.canvas.draw_idle()
284
285    fig.canvas.mpl_connect("key_press_event", press)
286
287    plt.show()
288
289    return fig

293def _drawDelayGraph(RayListAnalysed, Detector, Distance, DeltaFT, DrawAiryAndFourier=False, ColorCoded=None, fig=None):
294    """
295    Draws the 3D-delay-spot-diagram for a fixed detector. See more doc in the function below.
296    """
297    NumericalAperture = mp.ReturnNumericalAperture(RayListAnalysed, 1)  # NA determined from final ray bundle
298    Wavelength = RayListAnalysed[0].wavelength
299    AiryRadius = mp.ReturnAiryRadius(Wavelength, NumericalAperture) * 1e3  # in µm
300
301    DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD = _getDetectorPoints(
302        RayListAnalysed, Detector
303    )
304
305    DelayList = Detector.get_Delays(RayListAnalysed)
306    DurationSD = mp.StandardDeviation(DelayList)
307
308    if ColorCoded == "Intensity":
309        IntensityList = [k.intensity for k in RayListAnalysed]
310    elif ColorCoded == "Incidence":
311        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # in degree
312
313    plt.ion()
314    if fig is None:
315        fig = plt.figure()
316    else:
317        fig.clear(keep_observers=True)
318
319    ax = Axes3D(fig)
320    fig.add_axes(ax)
321    ax.set_xlabel("X (µm)")
322    ax.set_ylabel("Y (µm)")
323    ax.set_zlabel("Delay (fs)")
324
325    labelLine = (
326        "{:.3f}".format(Distance)
327        + " mm\n"
328        + "{:.1f}".format(SpotSizeSD * 1e3)
329        + " \u03BCm SD\n"
330        + "{:.2f}".format(DurationSD)
331        + " fs SD"
332    )
333    if ColorCoded == "Intensity":
334        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=IntensityList, label=labelLine)
335        # plt.title('Delay + Intensity graph\n press left/right to move detector position')
336        ax.set_title("Delay + Intensity graph\n press left/right to move detector position")
337
338    elif ColorCoded == "Incidence":
339        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=IncidenceList, label=labelLine)
340        # plt.title('Delay + Incidence graph\n press left/right to move detector position')
341        ax.set_title("Delay + Incidence graph\n press left/right to move detector position")
342    else:
343        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=DelayList, label=labelLine)
344        # plt.title('Delay graph\n press left/right to move detector position')
345        ax.set_title("Delay graph\n press left/right to move detector position")
346
347    ax.legend(loc="upper right")
348
349    if DrawAiryAndFourier:
350        x = np.linspace(-AiryRadius, AiryRadius, 40)
351        z = np.linspace(np.mean(DelayList) - DeltaFT * 0.5, np.mean(DelayList) + DeltaFT * 0.5, 40)
352        x, z = np.meshgrid(x, z)
353        y = np.sqrt(AiryRadius**2 - x**2)
354        ax.plot_wireframe(x, y, z, color="grey", alpha=0.1)
355        ax.plot_wireframe(x, -y, z, color="grey", alpha=0.1)
356
357    axisLim = 1.1 * max(AiryRadius, 0.5 * FocalSpotSize * 1000)
358    ax.set_xlim(-axisLim, axisLim)
359    ax.set_ylim(-axisLim, axisLim)
360    # ax.set_zlim(-axisLim/3*10, axisLim/3*10) #same scaling as spatial axes 
361    
362    plt.show()
363    fig.canvas.draw()
364
365    return fig, NumericalAperture, AiryRadius, FocalSpotSize

369def DelayGraph(
370    RayListAnalysed, Detector, DeltaFT: (int, float), DrawAiryAndFourier=False, ColorCoded=None
371) -> plt.Figure:
372    """
373    Produce a an interactive figure with a spot diagram resulting from the RayListAnalysed
374    hitting the Detector, with the ray-delays shown in the 3rd dimension.
375    The detector distance can be shifted with the left-right cursor keys.
376    If DrawAiryAndFourier is True, a cylinder is shown whose diameter is the Airy-spot-size and
377    whose height is the Fourier-limited pulse duration 'given by 'DeltaFT'.
378    The 'spots' can optionally be color-coded by specifying ColorCoded as ["Intensity","Incidence"].
379
380    Parameters
381    ----------
382        RayListAnalysed : list(Ray)
383            List of objects of the ModuleOpticalRay.Ray-class.
384
385        Detector : Detector
386            An object of the ModuleDetector.Detector-class.
387
388        DeltaFT : (int, float)
389            The Fourier-limited pulse duration. Just used as a reference to compare the temporal spread
390            induced by the ray-delays.
391
392        DrawAiryAndFourier : bool, optional
393            Whether to draw a cylinder showing the Airy-spot-size and Fourier-limited-duration.
394            The default is False.
395
396        ColorCoded : str, optional
397            Color-code the spots according to one of ["Intensity","Incidence"].
398            The default is None.
399
400    Returns
401    -------
402        fig : matlplotlib-figure-handle.
403            Shows the interactive figure.
404    """
405    Dist = Detector.get_distance()
406    fig, NumericalAperture, AiryRadius, FocalSpotSize = _drawDelayGraph(
407        RayListAnalysed, Detector, Dist, DeltaFT, DrawAiryAndFourier, ColorCoded
408    )
409
410    distStep = min(50, max(0.0005, round(FocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000))  # in mm
411
412    movingDetector = Detector.copy_detector()
413
414    def press(event):
415        nonlocal Dist, distStep, movingDetector, fig
416        if event.key == "right":
417            movingDetector.shiftByDistance(distStep)
418            Dist += distStep
419            ax = fig.axes[0]
420            cam = [ax.azim, ax.elev, ax._dist]
421            fig, sameNumericalAperture, sameAiryRadius, newFocalSpotSize = _drawDelayGraph(
422                RayListAnalysed, movingDetector, Dist, DeltaFT, DrawAiryAndFourier, ColorCoded, fig
423            )
424            ax = fig.axes[0]
425            ax.azim, ax.elev, ax._dist = cam
426        elif event.key == "left":
427            if Dist > 1.5 * distStep:
428                movingDetector.shiftByDistance(-distStep)
429                Dist -= distStep
430            else:
431                movingDetector.shiftToDistance(0.5 * distStep)
432                Dist = 0.5 * distStep
433            ax = fig.axes[0]
434            cam = [ax.azim, ax.elev, ax._dist]
435
436            fig, sameNumericalAperture, sameAiryRadius, newFocalSpotSize = _drawDelayGraph(
437                RayListAnalysed, movingDetector, Dist, DeltaFT, DrawAiryAndFourier, ColorCoded, fig
438            )
439            ax = fig.axes[0]
440            ax.azim, ax.elev, ax._dist = cam
441        else:
442            return fig
443        distStep = min(
444            50, max(0.0005, round(newFocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000)
445        )  # in mm
446
447    fig.canvas.mpl_connect("key_press_event", press)
448
449    return fig

453def MirrorProjection(OpticalChain, ReflectionNumber: int, Detector=None, ColorCoded=None) -> plt.Figure:
454    """
455    Produce a plot of the ray impact points on the optical element with index 'ReflectionNumber'.
456    The points can be color-coded according ["Incidence","Intensity","Delay"], where the ray delay is
457    measured at the Detector.
458
459    Parameters
460    ----------
461        OpticalChain : OpticalChain
462           List of objects of the ModuleOpticalOpticalChain.OpticalChain-class.
463
464        ReflectionNumber : int
465            Index specifying the optical element on which you want to see the impact points.
466
467        Detector : Detector, optional
468            Object of the ModuleDetector.Detector-class. Only necessary to project delays. The default is None.
469
470        ColorCoded : str, optional
471            Specifies which ray property to color-code: ["Incidence","Intensity","Delay"]. The default is None.
472
473    Returns
474    -------
475        fig : matlplotlib-figure-handle.
476            Shows the figure.
477    """
478    from mpl_toolkits.axes_grid1 import make_axes_locatable
479
480    Position = OpticalChain[ReflectionNumber].position
481    q = OpticalChain[ReflectionNumber].orientation
482    # n = OpticalChain.optical_elements[ReflectionNumber].normal
483    # m = OpticalChain.optical_elements[ReflectionNumber].majoraxis
484
485    RayListAnalysed = OpticalChain.get_output_rays()[ReflectionNumber]
486    # transform rays into the mirror-support reference frame
487    # (same as mirror frame but without the shift by mirror-centre)
488    RayList = mgeo.TranslationRayList(RayListAnalysed, -Position)
489    # RayList = mgeo.RotationRayList(RayList, n, np.array([0, 0, 1]))
490    # mPrime = mgeo.RotationPoint(m, n, np.array([0, 0, 1]))
491    # RayList = mgeo.RotationRayList(RayList, mPrime, np.array([1, 0, 0]))
492    RayList = mgeo.RotationRayListByQ(RayList, q)
493
494    x = np.asarray([k.point[0] for k in RayList])
495    y = np.asarray([k.point[1] for k in RayList])
496    if ColorCoded == "Intensity":
497        IntensityList = [k.intensity for k in RayListAnalysed]
498        z = np.asarray(IntensityList)
499        zlabel = "Intensity (arb.u.)"
500        title = "Ray intensity projected on mirror              "
501    elif ColorCoded == "Incidence":
502        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # in degree
503        z = np.asarray(IncidenceList)
504        zlabel = "Incidence angle (deg)"
505        title = "Ray incidence projected on mirror              "
506    elif ColorCoded == "Delay":
507        if Detector is not None:
508            z = np.asarray(Detector.get_Delays(RayListAnalysed))
509            zlabel = "Delay (fs)"
510            title = "Ray delay at detector projected on mirror              "
511        else:
512            raise ValueError("If you want to project ray delays, you must specify a detector.")
513    else:
514        z = "red"
515        title = "Ray impact points projected on mirror"
516
517    plt.ion()
518    fig = plt.figure()
519    ax = OpticalChain.optical_elements[ReflectionNumber].support._ContourSupport(fig)
520    p = plt.scatter(x, y, c=z, s=15)
521    if ColorCoded == "Delay" or ColorCoded == "Incidence" or ColorCoded == "Intensity":
522        divider = make_axes_locatable(ax)
523        cax = divider.append_axes("right", size="5%", pad=0.05)
524        cbar = fig.colorbar(p, cax=cax)
525        cbar.set_label(zlabel)
526    ax.set_xlabel("x (mm)")
527    ax.set_ylabel("y (mm)")
528    plt.title(title, loc="right")
529    plt.tight_layout()
530
531    bbox = ax.get_position()
532    bbox.set_points(bbox.get_points() - np.array([[0.01, 0], [0.01, 0]]))
533    ax.set_position(bbox)
534    plt.show()
535
536    return fig

540def _RenderRays(RayListHistory, EndDistance, maxRays=150, color_by_number = True):
541    meshes = []
542    # Ray display
543    for k in range(len(RayListHistory)):
544        x = []
545        y = []
546        z = []
547        if k != len(RayListHistory) - 1:
548            knums = list(
549                map(lambda x: x.number, RayListHistory[k])
550            )  # make a list of all ray numbers that are still in the game
551            if len(RayListHistory[k + 1]) > maxRays:
552                rays_to_render = np.random.choice(RayListHistory[k + 1], maxRays, replace=False)
553            else:
554                rays_to_render = RayListHistory[k + 1]
555
556            for j in rays_to_render:
557                indx = knums.index(j.number)
558                i = RayListHistory[k][indx]
559                Point1 = i.point
560                Point2 = j.point
561                x += [Point1[0], Point2[0]]
562                y += [Point1[1], Point2[1]]
563                z += [Point1[2], Point2[2]]
564
565        else:
566            if len(RayListHistory[k]) > maxRays:
567                rays_to_render = np.random.choice(RayListHistory[k], maxRays, replace=False)
568            else:
569                rays_to_render = RayListHistory[k]
570
571            for j in rays_to_render:
572                Point = j.point
573                Vector = j.vector
574                x += [Point[0], Point[0] + Vector[0] * EndDistance]
575                y += [Point[1], Point[1] + Vector[1] * EndDistance]
576                z += [Point[2], Point[2] + Vector[2] * EndDistance]
577        points = np.column_stack((x, y, z))
578        meshes += [pv.line_segments_from_points(points)]
579    return meshes

581def generate_distinct_colors(num_colors):
582    # Get a color palette from colorcet
583    palette = cc.glasbey
584
585    # Make sure the number of colors does not exceed the palette length
586    num_colors = min(num_colors, len(palette))
587
588    # Slice the palette to get the desired number of colors
589    distinct_colors = palette[:num_colors]
590
591    return distinct_colors

594    """
