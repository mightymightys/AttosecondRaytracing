 26def _getDetectorPoints(RayListAnalysed, Detector) -> tuple[np.ndarray, np.ndarray, float, float]:
 27    """
 28    Prepare the ray impact points on the detector in a format used for the plotting,
 29    and along the way also calculate the "spotsize" of this point-cloud on the detector.
 30
 31    Parameters
 32    ----------
 33        RayListAnalysed : list(Ray)
 34            A list of objects of the ModuleOpticalRay.Ray-class.
 35
 36        Detector : Detector
 37            An object of the ModuleDetector.Detector-class.
 38
 39    Returns
 40    -------
 41        DectectorPoint2D_Xcoord : np.ndarray with x-coordinates
 42
 43        DectectorPoint2D_Ycoord : np.ndarray with y-coordinates
 44
 45        FocalSpotSize : float
 46
 47        SpotSizeSD : float
 48    """
 49
 50    DetectorPointList2DCentre = Detector.get_PointList2DCentre(RayListAnalysed)
 51    DectectorPoint2D_Xcoord = np.array([k[0] * 1e3 for k in DetectorPointList2DCentre])  # mm to Âµm
 52    DectectorPoint2D_Ycoord = np.array([k[1] * 1e3 for k in DetectorPointList2DCentre])
 53
 54    FocalSpotSize = mgeo.DiameterPointList(DetectorPointList2DCentre)
 55    SpotSizeSD = mp.StandardDeviation(DetectorPointList2DCentre)
 56    return DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD

 60def getETransmission(RayListIn, RayListOut) -> float:
 61    """
 62    Calculates the energy transmission from RayListIn to RayListOut in percent by summing up the
 63    intensity-property of the individual rays.
 64
 65    Parameters
 66    ----------
 67        RayListIn, RayListOut : list(Ray)
 68            Lists of objects of the ModuleOpticalRay.Ray-class.
 69
 70    Returns
 71    -------
 72        ETransmission : float
 73    """
 74    ETransmission = 100 * sum(Ray.intensity for Ray in RayListOut) / sum(Ray.intensity for Ray in RayListIn)
 75    return ETransmission

 79def GetResultSummary(Detector, RayListAnalysed, verbose=False):
 80    """
 81    Calculate and return FocalSpotSize-standard-deviation and Duration-standard-deviation
 82    for the given Detector and RayList.
 83    If verbose, then also print a summary of the results for the given Detector.
 84
 85    Parameters
 86    ----------
 87        Detector : Detector
 88            An object of the ModuleDetector.Detector-class.
 89
 90        RayListAnalysed : list(Ray)
 91            List of objects of the ModuleOpticalRay.Ray-class.
 92
 93        verbose : bool
 94            Whether to print a result summary.
 95
 96    Returns
 97    -------
 98        FocalSpotSizeSD : float
 99
100        DurationSD : float
101    """
102    DetectorPointList2DCentre = Detector.get_PointList2DCentre(RayListAnalysed)
103    FocalSpotSizeSD = mp.StandardDeviation(DetectorPointList2DCentre)
104    DelayList = Detector.get_Delays(RayListAnalysed)
105    DurationSD = mp.StandardDeviation(DelayList)
106
107    if verbose:
108        FocalSpotSize = mgeo.DiameterPointList(DetectorPointList2DCentre)
109        summarystring = (
110            "At the detector distance of "
111            + "{:.3f}".format(Detector.get_distance())
112            + " mm we get:\n"
113            + "Spatial std : "
114            + "{:.3f}".format(FocalSpotSizeSD * 1e3)
115            + " \u03BCm and min-max: "
116            + "{:.3f}".format(FocalSpotSize * 1e3)
117            + " \u03BCm\n"
118            + "Temporal std : "
119            + "{:.3e}".format(DurationSD)
120            + " fs and min-max : "
121            + "{:.3e}".format(max(DelayList) - min(DelayList))
122            + " fs"
123        )
124
125        print(summarystring)
126
127    return FocalSpotSizeSD, DurationSD

131def SpotDiagram(RayListAnalysed, Detector, DrawAiryAndFourier=False, ColorCoded=None) -> plt.Figure:
132    """
133    Produce a an interactive figure with the spot diagram resulting from the RayListAnalysed
134    hitting the Detector.
135    The detector distance can be shifted with the left-right cursor keys.
136    If DrawAiryAndFourier is True, a circle with the Airy-spot-size will be shown.
137    The 'spots' can optionally be color-coded by specifying ColorCoded
138    as ["Intensity","Incidence","Delay"].
139
140    Parameters
141    ----------
142        RayListAnalysed : list(Ray)
143            List of objects of the ModuleOpticalRay.Ray-class.
144
145        Detector : Detector
146            An object of the ModuleDetector.Detector-class.
147
148        DrawAiryAndFourier : bool, optional
149            Whether to draw a circle with the Airy-spot-size. The default is False.
150
151        ColorCoded : str, optional
152            Color-code the spots according to one of ["Intensity","Incidence","Delay"]. The default is None.
153
154    Returns
155    -------
156        fig : matlplotlib-figure-handle.
157            Shows the interactive figure.
158    """
159    NumericalAperture = mp.ReturnNumericalAperture(RayListAnalysed, 1)  # NA determined from final ray bundle
160    Wavelength = RayListAnalysed[0].wavelength
161    if DrawAiryAndFourier:
162        AiryRadius = mp.ReturnAiryRadius(Wavelength, NumericalAperture) * 1e3  # in Âµm
163    else:
164        AiryRadius = 0
165
166    DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD = _getDetectorPoints(
167        RayListAnalysed, Detector
168    )
169
170    if ColorCoded == "Intensity":
171        IntensityList = [k.intensity for k in RayListAnalysed]
172        z = np.asarray(IntensityList)
173        zlabel = "Intensity (arb.u.)"
174        title = "Intensity + Spot Diagram\n press left/right to move detector position"
175        addLine = ""
176    elif ColorCoded == "Incidence":
177        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # degree
178        z = np.asarray(IncidenceList)
179        zlabel = "Incidence angle (deg)"
180        title = "Ray Incidence + Spot Diagram\n press left/right to move detector position"
181        addLine = ""
182    elif ColorCoded == "Delay":
183        DelayList = Detector.get_Delays(RayListAnalysed)
184        DurationSD = mp.StandardDeviation(DelayList)
185        z = np.asarray(DelayList)
186        zlabel = "Delay (fs)"
187        title = "Delay + Spot Diagram\n press left/right to move detector position"
188        addLine = "\n" + "{:.2f}".format(DurationSD) + " fs SD"
189    else:
190        z = "red"
191        title = "Spot Diagram\n press left/right to move detector position"
192        addLine = ""
193
194    Dist = Detector.get_distance()
195    distStep = min(50, max(0.0005, round(FocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000))  # in mm
196
197    plt.ion()
198    fig, ax = plt.subplots()
199    if DrawAiryAndFourier:
200        theta = np.linspace(0, 2 * np.pi, 100)
201        x = AiryRadius * np.cos(theta)
202        y = AiryRadius * np.sin(theta)  #
203        ax.plot(x, y, c="black")
204
205    foo = ax.scatter(
206        DectectorPoint2D_Xcoord,
207        DectectorPoint2D_Ycoord,
208        c=z,
209        s=15,
210        label="{:.3f}".format(Dist) + " mm\n" + "{:.1f}".format(SpotSizeSD * 1e3) + " \u03BCm SD" + addLine,
211    )
212
213    axisLim = 1.1 * max(AiryRadius, 0.5 * FocalSpotSize * 1000)
214    ax.set_xlim(-axisLim, axisLim)
215    ax.set_ylim(-axisLim, axisLim)
216
217    if ColorCoded == "Intensity" or ColorCoded == "Incidence" or ColorCoded == "Delay":
218        cbar = fig.colorbar(foo)
219        cbar.set_label(zlabel)
220
221    ax.legend(loc="upper right")
222    ax.set_xlabel("X (Âµm)")
223    ax.set_ylabel("Y (Âµm)")
224    ax.set_title(title)
225    # ax.margins(x=0)
226
227    movingDetector = Detector.copy_detector()
228
229    def press(event):
230        nonlocal Dist, distStep, movingDetector, ColorCoded, zlabel, cbar
231        if event.key == "right":
232            movingDetector.shiftByDistance(distStep)
233            Dist += distStep
234        elif event.key == "left":
235            if Dist > 1.5 * distStep:
236                movingDetector.shiftByDistance(-distStep)
237                Dist -= distStep
238            else:
239                movingDetector.shiftToDistance(0.5 * distStep)
240                Dist = 0.5 * distStep
241        else:
242            return None
243        newDectectorPoint2D_Xcoord, newDectectorPoint2D_Ycoord, newFocalSpotSize, newSpotSizeSD = _getDetectorPoints(
244            RayListAnalysed, movingDetector
245        )
246        xy = foo.get_offsets()
247        xy[:, 0] = newDectectorPoint2D_Xcoord
248        xy[:, 1] = newDectectorPoint2D_Ycoord
249        foo.set_offsets(xy)
250
251        if ColorCoded == "Delay":
252            newDelayList = np.asarray(movingDetector.get_Delays(RayListAnalysed))
253            newDurationSD = mp.StandardDeviation(newDelayList)
254            newaddLine = "\n" + "{:.2f}".format(newDurationSD) + " fs SD"
255            foo.set_array(newDelayList)
256            foo.set_clim(min(newDelayList), max(newDelayList))
257            cbar.update_normal(foo)
258
259        foo.set_label(
260            "{:.3f}".format(Dist) + " mm\n" + "{:.1f}".format(newSpotSizeSD * 1e3) + " \u03BCm SD" + newaddLine
261        )
262        ax.legend(loc="upper right")
263
264        axisLim = 1.1 * max(AiryRadius, 0.5 * newFocalSpotSize * 1000)
265        ax.set_xlim(-axisLim, axisLim)
266        ax.set_ylim(-axisLim, axisLim)
267
268        distStep = min(
269            50, max(0.0005, round(newFocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000)
270        )  # in mm
271
272        fig.canvas.draw_idle()
273
274    fig.canvas.mpl_connect("key_press_event", press)
275
276    plt.show()
277
278    return fig

282def _drawDelayGraph(RayListAnalysed, Detector, Distance, DeltaFT, DrawAiryAndFourier=False, ColorCoded=None, fig=None):
283    """
284    Draws the 3D-delay-spot-diagram for a fixed detector. See more doc in the function below.
285    """
286    NumericalAperture = mp.ReturnNumericalAperture(RayListAnalysed, 1)  # NA determined from final ray bundle
287    Wavelength = RayListAnalysed[0].wavelength
288    AiryRadius = mp.ReturnAiryRadius(Wavelength, NumericalAperture) * 1e3  # in Âµm
289
290    DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, FocalSpotSize, SpotSizeSD = _getDetectorPoints(
291        RayListAnalysed, Detector
292    )
293
294    DelayList = Detector.get_Delays(RayListAnalysed)
295    DurationSD = mp.StandardDeviation(DelayList)
296
297    if ColorCoded == "Intensity":
298        IntensityList = [k.intensity for k in RayListAnalysed]
299    elif ColorCoded == "Incidence":
300        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # in degree
301
302    plt.ion()
303    if fig is None:
304        fig = plt.figure()
305    else:
306        fig.clear(keep_observers=True)
307
308    ax = Axes3D(fig)
309    fig.add_axes(ax)
310    ax.set_xlabel("X (Âµm)")
311    ax.set_ylabel("Y (Âµm)")
312    ax.set_zlabel("Delay (fs)")
313
314    labelLine = (
315        "{:.3f}".format(Distance)
316        + " mm\n"
317        + "{:.1f}".format(SpotSizeSD * 1e3)
318        + " \u03BCm SD\n"
319        + "{:.2f}".format(DurationSD)
320        + " fs SD"
321    )
322    if ColorCoded == "Intensity":
323        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=IntensityList, label=labelLine)
324        # plt.title('Delay + Intensity graph\n press left/right to move detector position')
325        ax.set_title("Delay + Intensity graph\n press left/right to move detector position")
326
327    elif ColorCoded == "Incidence":
328        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=IncidenceList, label=labelLine)
329        # plt.title('Delay + Incidence graph\n press left/right to move detector position')
330        ax.set_title("Delay + Incidence graph\n press left/right to move detector position")
331    else:
332        ax.scatter(DectectorPoint2D_Xcoord, DectectorPoint2D_Ycoord, DelayList, s=4, c=DelayList, label=labelLine)
333        # plt.title('Delay graph\n press left/right to move detector position')
334        ax.set_title("Delay graph\n press left/right to move detector position")
335
336    ax.legend(loc="upper right")
337
338    if DrawAiryAndFourier:
339        x = np.linspace(-AiryRadius, AiryRadius, 40)
340        z = np.linspace(np.mean(DelayList) - DeltaFT * 0.5, np.mean(DelayList) + DeltaFT * 0.5, 40)
341        x, z = np.meshgrid(x, z)
342        y = np.sqrt(AiryRadius**2 - x**2)
343        ax.plot_wireframe(x, y, z, color="grey", alpha=0.1)
344        ax.plot_wireframe(x, -y, z, color="grey", alpha=0.1)
345
346    axisLim = 1.1 * max(AiryRadius, 0.5 * FocalSpotSize * 1000)
347    ax.set_xlim(-axisLim, axisLim)
348    ax.set_ylim(-axisLim, axisLim)
349
350    plt.show()
351    fig.canvas.draw()
352
353    return fig, NumericalAperture, AiryRadius, FocalSpotSize

357def DelayGraph(
358    RayListAnalysed, Detector, DeltaFT: (int, float), DrawAiryAndFourier=False, ColorCoded=None
359) -> plt.Figure:
360    """

365    whose height is the Fourier-limited pulse duration 'given by 'DeltaFT'.
366    The 'spots' can optionally be color-coded by specifying ColorCoded as ["Intensity","Incidence"].
367
368    Parameters
369    ----------

371            List of objects of the ModuleOpticalRay.Ray-class.
372
373        Detector : Detector
374            An object of the ModuleDetector.Detector-class.
375

377            The Fourier-limited pulse duration. Just used as a reference to compare the temporal spread
378            induced by the ray-delays.
379
380        DrawAiryAndFourier : bool, optional
381            Whether to draw a cylinder showing the Airy-spot-size and Fourier-limited-duration.
382            The default is False.
383
384        ColorCoded : str, optional
385            Color-code the spots according to one of ["Intensity","Incidence"].

387

399
400    movingDetector = Detector.copy_detector()
401
402    def press(event):
403        nonlocal Dist, distStep, movingDetector, fig

414        elif event.key == "left":

417                Dist -= distStep
418            else:
419                movingDetector.shiftToDistance(0.5 * distStep)
420                Dist = 0.5 * distStep
421            ax = fig.axes[0]
422            cam = [ax.azim, ax.elev, ax.dist]
423
424            fig, sameNumericalAperture, sameAiryRadius, newFocalSpotSize = _drawDelayGraph(
425                RayListAnalysed, movingDetector, Dist, DeltaFT, DrawAiryAndFourier, ColorCoded, fig

427            ax = fig.axes[0]

430            return fig
431        distStep = min(
432            50, max(0.0005, round(newFocalSpotSize / 8 / np.arcsin(NumericalAperture) * 10000) / 10000)
433        )  # in mm
434
435    fig.canvas.mpl_connect("key_press_event", press)
436
437    return fig

441def MirrorProjection(OpticalChain, ReflectionNumber: int, Detector=None, ColorCoded=None) -> plt.Figure:
442    """
443    Produce a plot of the ray impact points on the optical element with index 'ReflectionNumber'.
444    The points can be color-coded according ["Incidence","Intensity","Delay"], where the ray delay is
445    measured at the Detector.
446
447    Parameters
448    ----------
449        OpticalChain : OpticalChain
450           List of objects of the ModuleOpticalOpticalChain.OpticalChain-class.
451
452        ReflectionNumber : int
453            Index specifying the optical element on which you want to see the impact points.
454
455        Detector : Detector, optional
456            Object of the ModuleDetector.Detector-class. Only necessary to project delays. The default is None.
457
458        ColorCoded : str, optional
459            Specifies which ray property to color-code: ["Incidence","Intensity","Delay"]. The default is None.
460
461    Returns
462    -------
463        fig : matlplotlib-figure-handle.
464            Shows the figure.
465    """
466    from mpl_toolkits.axes_grid1 import make_axes_locatable
467
468    Position = OpticalChain.optical_elements[ReflectionNumber].position
469    n = OpticalChain.optical_elements[ReflectionNumber].normal
470    m = OpticalChain.optical_elements[ReflectionNumber].majoraxis
471
472    RayListAnalysed = OpticalChain.get_output_rays()[ReflectionNumber]
473    # transform rays into the mirror-support reference frame
474    # (same as mirror frame but without the shift by mirror-centre)
475    RayList = mgeo.TranslationRayList(RayListAnalysed, -Position)
476    RayList = mgeo.RotationRayList(RayList, n, np.array([0, 0, 1]))
477    mPrime = mgeo.RotationPoint(m, n, np.array([0, 0, 1]))
478    RayList = mgeo.RotationRayList(RayList, mPrime, np.array([1, 0, 0]))
479
480    x = np.asarray([k.point[0] for k in RayList])
481    y = np.asarray([k.point[1] for k in RayList])
482    if ColorCoded == "Intensity":
483        IntensityList = [k.intensity for k in RayListAnalysed]
484        z = np.asarray(IntensityList)
485        zlabel = "Intensity (arb.u.)"
486        title = "Ray intensity projected on mirror              "
487    elif ColorCoded == "Incidence":
488        IncidenceList = [np.rad2deg(k.incidence) for k in RayListAnalysed]  # in degree
489        z = np.asarray(IncidenceList)
490        zlabel = "Incidence angle (deg)"
491        title = "Ray incidence projected on mirror              "
492    elif ColorCoded == "Delay":
493        if Detector is not None:
494            z = np.asarray(Detector.get_Delays(RayListAnalysed))
495            zlabel = "Delay (fs)"
496            title = "Ray delay at detector projected on mirror              "
497        else:
498            raise ValueError("If you want to project ray delays, you must specify a detector.")
499    else:
500        z = "red"
501        title = "Ray impact points projected on mirror"
502
503    plt.ion()
504    fig = plt.figure()
505    ax = OpticalChain.optical_elements[ReflectionNumber].type.support._ContourSupport(fig)
506    p = plt.scatter(x, y, c=z, s=15)
507    if ColorCoded == "Delay" or ColorCoded == "Incidence" or ColorCoded == "Intensity":
508        divider = make_axes_locatable(ax)
509        cax = divider.append_axes("right", size="5%", pad=0.05)
510        cbar = fig.colorbar(p, cax=cax)
511        cbar.set_label(zlabel)
512    ax.set_xlabel("x (mm)")
513    ax.set_ylabel("y (mm)")
514    plt.title(title, loc="right")
515    plt.tight_layout()
516
517    bbox = ax.get_position()
518    bbox.set_points(bbox.get_points() - np.array([[0.01, 0], [0.01, 0]]))
519    ax.set_position(bbox)
520    plt.show()
521
522    return fig

526def RayRenderGraph(
527    OpticalChain,
528    EndDistance=None,
529    maxRays=300,
530    OEpoints=3000,
531    scale_spheres=0.5,
532    tube_width=0.05,
533    slow_method=False,
534    draw_mesh=False,
535):
536    """
537    Renders an image of the Optical setup and the traced rays.
538
539    Parameters
540    ----------
541        OpticalChain : OpticalChain
542            List of objects of the ModuleOpticalOpticalChain.OpticalChain-class.
543
544        EndDistance : float, optional
545            The rays of the last ray bundle are drawn with a length given by EndDistance (in mm). If not specified,
546            this distance is set to that between the source point and the 1st optical element.
547
548        maxRays: int
549            The maximum number of rays to render. Rendering all the traced rays is a insufferable resource hog
550            and not required for a nice image. Default is 150.
551
552        OEpoints : int
553            How many little spheres to draw to represent the optical elements.  Default is 2000.
554
555    Returns
556    -------
557        fig : mayavi-figure-handle.
558            Shows the figure.
559    """
560
561    RayListHistory = [OpticalChain.source_rays] + OpticalChain.get_output_rays()
562
563    if EndDistance is None:
564        EndDistance = np.linalg.norm(OpticalChain.source_rays[0].point - OpticalChain.optical_elements[0].position)
565
566    print("...rendering image of optical chain...", end="", flush=True)
567    fig = mlab.figure(bgcolor=(1, 1, 1), size=(1500, 500))
568
569    x = []
570    y = []
571    z = []
572    # Ray display
573    for k in range(len(RayListHistory)):
574        if k != len(RayListHistory) - 1:
575            knums = list(
576                map(lambda x: x.number, RayListHistory[k])
577            )  # make a list of all ray numbers that are still in the game
578            if len(RayListHistory[k + 1]) > maxRays:
579                rays_to_render = np.random.choice(RayListHistory[k + 1], maxRays, replace=False)
580            else:
581                rays_to_render = RayListHistory[k + 1]
582
583            for j in rays_to_render:
584                indx = knums.index(j.number)
585                i = RayListHistory[k][indx]
586                Point1 = i.point
587                Point2 = j.point
588                x += [Point1[0], Point2[0]]
589                y += [Point1[1], Point2[1]]
590                z += [Point1[2], Point2[2]]
591                if slow_method:
592                    mlab.plot3d(x[-2:], y[-2:], z[-2:], color=(1, 0, 0), tube_radius=0.05)
593
594        else:
595            if len(RayListHistory[k]) > maxRays:
